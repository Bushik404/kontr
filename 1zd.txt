#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include <string.h>
#define NUMS 13

char* to_roman( int num, char* buff )
{
int p = NUMS;
unsigned int a_num[NUMS] = {1,4,5,9,10,40,50,90,100,400,500,900,1000};
char* r_num[NUMS] = {"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"};
char* roman;

buff = '\0';
while (p--)
while (num >= a_num[p])
{
num -= a_num[p];
strcat(buff, r_num[p]);
}

return buff;
}

int main( void)
{
char roman;
int arabian;
scanf("%d",&arabian);
roman = (char*) malloc (256);
printf( "%i: %s\n", arabian, to_roman( arabian, roman ));
}
